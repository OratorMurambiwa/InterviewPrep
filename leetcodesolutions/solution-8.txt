class Solution(object):
    def findDiagonalOrder(self, nums):
        """
        :type nums: List[List[int]]
        :rtype: List[int]
        """
        groups = {}

        for r in range(len(nums)):
            for c in range(len(nums[r])):
                sumd = r + c
                if sumd not in groups:
                    groups[sumd] = []
                groups[sumd].append(nums[r][c])

        res = []
        for sumd in sorted(groups.keys()):
            res.extend(groups[sumd][::-1])
        return res
