class Solution(object):
    def numPairsDivisibleBy60(self, time):
        """
        :type time: List[int]
        :rtype: int
        """
        freq =defaultdict(int)
        pairs = 0
 

        for i in time:
            r = i % 60
            complement = (60-r) % 60

            pairs += freq[complement]
            freq[r] += 1

        return pairs
