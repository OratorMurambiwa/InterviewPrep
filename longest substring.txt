class Solution(object):
    def lengthOfLongestSubstring(self, s):
        """
        :type s: str
        :rtype: int
        """
        if len(s) == 0:
            return 0

        seen = {}
        left = 0
        longest = 0

        for right in range(len(s)):
            if s[right] in seen:
                left = max(left, seen[s[right]] + 1)

            longest = max(longest, right - left + 1)
            seen[s[right]] = right

        return longest
